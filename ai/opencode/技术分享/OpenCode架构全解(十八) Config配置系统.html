二、7 级优先级加载顺序

配置按从低到高的优先级加载，高优先级的同名字段覆盖低优先级：

// config.ts — state() 中的加载顺序
let result: Info = {}

// 1. Remote .well-known/opencode（组织级远程默认）
for (const [key, value] of Object.entries(auth)) {
    if (value.type === "wellknown") {
        const response = await fetch(`${url}/.well-known/opencode`)
        result = mergeConfigConcatArrays(result, await load(response))
    }
}

// 2. Global user config（~/.config/opencode/）
result = mergeConfigConcatArrays(result, await global())

// 3. Custom config（OPENCODE_CONFIG 环境变量）
if (Flag.OPENCODE_CONFIG) {
    result = mergeConfigConcatArrays(result, await loadFile(Flag.OPENCODE_CONFIG))
}

// 4. Project config（项目目录下的 opencode.json / opencode.jsonc）
if (!Flag.OPENCODE_DISABLE_PROJECT_CONFIG) {
    for (const file of await ConfigPaths.projectFiles("opencode", ...)) {
        result = mergeConfigConcatArrays(result, await loadFile(file))
    }
}

// 5. .opencode 目录（.opencode/agents/、.opencode/commands/、.opencode/plugins/、.opencode/opencode.json）
for (const dir of unique(directories)) {
    result = mergeConfigConcatArrays(result, await loadFile(path.join(dir, "opencode.json")))
    result.command = mergeDeep(result.command, await loadCommand(dir))
    result.agent = mergeDeep(result.agent, await loadAgent(dir))
    result.plugin.push(...(await loadPlugin(dir)))
}

// 6. Inline config（OPENCODE_CONFIG_CONTENT 环境变量）
if (process.env.OPENCODE_CONFIG_CONTENT) {
    result = mergeConfigConcatArrays(result, await load(OPENCODE_CONFIG_CONTENT))
}

// 7. Managed config（企业管理员目录 — 最高优先级）
if (existsSync(managedDir)) {
    for (const file of ["opencode.jsonc", "opencode.json"]) {
        result = mergeConfigConcatArrays(result, await loadFile(path.join(managedDir, file)))
    }
}

三、mergeConfigConcatArrays — 智能合并

普通的 mergeDeep 会替换数组字段，但 plugin 和 instructions 应该合并：

function mergeConfigConcatArrays(target: Info, source: Info): Info {
    const merged = mergeDeep(target, source)

    // plugin 数组：合并 + 去重
    if (target.plugin && source.plugin) {
        merged.plugin = Array.from(new Set([...target.plugin, ...source.plugin]))
    }

    // instructions 数组：合并 + 去重
    if (target.instructions && source.instructions) {
        merged.instructions = Array.from(new Set([...target.instructions, ...source.instructions]))
    }

    return merged
}
为什么不直接用 mergeDeep？remeda 的 mergeDeep 对数组字段默认是替换。但用户在全局配置中声明了 ["plugin-a"]，在项目配置中声明了 ["plugin-b"]，期望结果是两个插件都加载，而非只加载 plugin-b。

五、Permission 权限配置解析

权限配置是最复杂的部分，支持三种粒度的写法：

// 粒度 1：全局字符串 — 所有工具统一策略
"permission": "allow"

// 粒度 2：工具级 — 按工具名设置
"permission": {
    "read": "allow",
    "edit": "ask",
    "bash": "deny"
}

// 粒度 3：文件模式级 — 特定文件特殊处理
"permission": {
    "read": {
        "*.env": "ask",
        "*.ts": "allow"
    },
    "edit": {
        "src/**": "allow",
        "config/**": "ask"
    }
}

Zod 预处理保留键序

// 权限规则匹配依赖键的顺序！
// 用户写的顺序就是匹配顺序

const permissionPreprocess = (val: unknown) => {
    if (typeof val === "object" && val !== null && !Array.isArray(val)) {
        return { __originalKeys: Object.keys(val), ...val }
    }
    return val
}

const permissionTransform = (x: unknown) => {
    const { __originalKeys, ...rest } = x
    // 按原始键序重建对象
    const result = {}
    for (const key of __originalKeys) {
        if (key in rest) result[key] = rest[key]
    }
    return result
}

// Permission Schema：preprocess → parse → transform
export const Permission = z.preprocess(
    permissionPreprocess,
    z.object({
        read: PermissionRule.optional(),
        edit: PermissionRule.optional(),
        bash: PermissionRule.optional(),
        // ... 更多工具
    }).catchall(PermissionRule)  // 支持自定义工具名
)
.transform(permissionTransform)

Zod 默认会按 schema 定义顺序排列键，但权限匹配依赖用户定义的顺序（例如 *.env 应该在 * 前面匹配）。通过 __originalKeys 技巧保留了原始顺序。

七、JSONC 解析与 $schema 自动注入

配置文件支持 JSONC 格式（JSON with Comments），并在首次加载时自动注入 $schema：

function parseConfig(text: string, filepath: string): Info {
    const errors: JsoncParseError[] = []
    const data = parseJsonc(text, errors, { allowTrailingComma: true })

    if (errors.length) {
        // 精确到行号和列号的错误提示
        const line = beforeOffset.length
        const column = beforeOffset[beforeOffset.length - 1].length + 1
        throw new JsonError({
            path: filepath,
            message: `\nLine ${line}: ${problemLine}\n${"".padStart(column + 9)}^`,
        })
    }

    return Info.safeParse(data)
}

// 自动注入 $schema 方便 IDE 提示
if (!parsed.data.$schema && isFile) {
    parsed.data.$schema = "https://opencode.ai/config.json"
    const updated = original.replace(
        /^\s*\{/,
        '{\n  "$schema": "https://opencode.ai/config.json",'
    )
    await Filesystem.write(options.path, updated)
}

八、update / updateGlobal — 配置写回

系统不仅能读取配置，还能将修改写回文件：

// 项目级更新
export async function update(config: Info) {
    const filepath = path.join(Instance.directory, "config.json")
    const existing = await loadFile(filepath)
    await Filesystem.writeJson(filepath, mergeDeep(existing, config))
    await Instance.dispose()  // 重建 Instance state
}

// 全局配置更新（支持 JSONC 格式保留）
export async function updateGlobal(config: Info) {
    const filepath = globalConfigFile()
    const before = await Filesystem.readText(filepath)

    if (filepath.endsWith(".jsonc")) {
        // JSONC：使用 jsonc-parser 的 modify API 保留注释和格式
        const updated = patchJsonc(before, config)
        await Filesystem.write(filepath, updated)
    } else {
        // JSON：直接 mergeDeep + writeJson
        const existing = parseConfig(before, filepath)
        await Filesystem.writeJson(filepath, mergeDeep(existing, config))
    }

    global.reset()  // 清除缓存
    void Instance.disposeAll()  // 重建所有 Instance
}

patchJsonc — 格式保留的 JSONC 修改

function patchJsonc(input: string, patch: unknown, path: string[] = []): string {
    if (!isRecord(patch)) {
        // 叶子节点：使用 jsonc-parser 的 modify 编辑
        const edits = modify(input, path, patch, {
            formattingOptions: { insertSpaces: true, tabSize: 2 },
        })
        return applyEdits(input, edits)
    }

    // 递归处理嵌套对象
    return Object.entries(patch).reduce((result, [key, value]) => {
        return patchJsonc(result, value, [...path, key])
    }, input)
}

使用 jsonc-parser 库的 modify API，修改 JSONC 文件时保留原有注释、缩进和格式。

九、企业级 Managed Config

Managed Config 是最高优先级的配置源，由企业 IT 管理员控制，普通用户无法修改：

function systemManagedConfigDir(): string {
    switch (process.platform) {
        case "darwin":
            return "/Library/Application Support/opencode"
        case "win32":
            return path.join(process.env.ProgramData || "C:\\ProgramData", "opencode")
        default:
            return "/etc/opencode"
    }
}

// 最后加载 = 最高优先级
if (existsSync(managedDir)) {
    for (const file of ["opencode.jsonc", "opencode.json"]) {
        result = mergeConfigConcatArrays(result, await loadFile(path.join(managedDir, file)))
    }
}
Managed 与普通目录的区别：Managed 目录只加载 opencode.json 文件，不会扫描 agents/、commands/、plugins/ 子目录。这是有意为之 — 系统目录需要 elevated 权限，在这些目录下执行 npm install 会失败。因此 Managed 只负责配置文件覆盖，不参与插件安装。

十、依赖安装与版本检查

.opencode/ 目录可以有自己的 package.json，系统自动安装依赖：

export async function needsInstall(dir: string) {
    const nodeModules = path.join(dir, "node_modules")
    if (!existsSync(nodeModules)) return true

    const pkg = path.join(dir, "package.json")
    const parsed = await Filesystem.readJson(pkg)
    const depVersion = parsed.dependencies["@opencode-ai/plugin"]

    if (!depVersion) return true
    if (targetVersion === "latest") {
        // 检查 npm registry 是否有更新版本
        return await PackageRegistry.isOutdated("@opencode-ai/plugin", depVersion, dir)
    }
    if (depVersion === targetVersion) return false
    return true
}

export async function installDependencies(dir: string) {
    const targetVersion = Installation.isLocal() ? "*" : Installation.VERSION

    // 写入 package.json（确保 @opencode-ai/plugin 依赖存在）
    json.dependencies = {
        ...json.dependencies,
        "@opencode-ai/plugin": targetVersion,
    }
    await Filesystem.writeJson(pkg, json)

    // 写入 .gitignore（避免提交自动生成的文件）
    if (!hasGitIgnore)
        await Filesystem.write(gitignore,
            ["node_modules", "package.json", "bun.lock", ".gitignore"].join("\n")
        )

    // bun install
    await BunProc.run(["install", ...(proxied() ? ["--no-cache"] : [])], { cwd: dir })
}
只读目录自动跳过：系统通过 fs.access(dir, W_OK) 检查目录是否可写。不可写的配置目录（如某些全局目录）会跳过依赖安装，只加载配置文件本身。

插件去重

export function deduplicatePlugins(plugins: string[]): string[] {
    const seenNames = new Set()
    // 反向遍历 → 高优先级的先被记录
    for (const specifier of plugins.toReversed()) {
        const name = getPluginName(specifier)
        if (!seenNames.has(name)) {
            seenNames.add(name)
            uniqueSpecifiers.push(specifier)
        }
    }
    return uniqueSpecifiers.toReversed()
}
